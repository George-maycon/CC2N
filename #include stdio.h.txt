 // tratamento de erro
 if(X < 1)
    printf("ERRO: Valor incorreto do Numero\n");
else{
    printf("Os impares entre [%i, %.0f] sao:\n", X, pow(X,3));
    for(contador = X; contador <= pow(X,3);contador++){
        //impares
        if(contador % 2 != 0){
                //enumerar
            quantidade++;
            //
            soma = soma + contador;
            printf("impar %i: %i\n", quantidade, contador);

        }
    }
        media = 1.0 * soma / quantidade;

}

    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#define SM 1518.01
int main()
{
    int aluno = 0;   //start
    float salario, soma = 0, media;
    printf("PESQUISA SALARIAL UNIVERSITARIA: \n");
    while(aluno < 3){ //stop

        printf("ALUNO %i - SALARIO (>= R$ %.2f): ", aluno + 1, SM);
        scanf ("%f", &salario);
        //TRATAMENTO DE ERRO:
        if(salario < SM)
            printf("ERRo: escolha de novo.\n");
        else{
            soma = soma + salario;
            aluno++;
        }

    }
media = soma / aluno;
printf("MEDIA SALARIAL: R$%.2f\n", media);
    return 0;
}
